import java.util.*;
public class GFG {

	{

		set.add("practice");
		set.add("contribute");
		set.add("ide");
		System.out.println("Original HashSet: "+ set);

		Collections.sort(list);
		System.out.println("HashSet elements "+ "in sorted order "+ "using List: "+ list);
	}
}
